async function buyPlan(planId) { try { const res = await fetch("https://stripe-backend-hu55.onrender.com/create-checkout-session", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ planId }) }); const data = await res.json(); if (data.url) { window.location.href = data.url; } else { alert("Payment failed: " + (data.error || "Unknown error")); } } catch (err) { alert("Request failed"); console.error(err); } }
